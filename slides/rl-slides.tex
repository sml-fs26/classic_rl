    \only<4>{\Large $q(s_i, a_i) \leftarrow \underbrace{q(s_i, a_i)}_{\text{old guess}} + \underbrace{\alpha}_{0.15} \bigl[\, \underbrace{r_i}_{\text{what the week paid}} - \ q(s_i, a_i) \,\bigr]$}%
    \only<5->{\Large $q(s_i, a_i) \leftarrow \underbrace{q(s_i, a_i)}_{\text{old guess}} + \underbrace{\alpha}_{0.15} \underbrace{\bigl[\, r_i - q(s_i, a_i) \,\bigr]}_{\text{the miss: move a fraction toward it}}$}%
  }
  \formulaslot{0.7cm}{%
    \uncover<6->{{\color{muted}exponential smoothing: new forecast $=$ old forecast $+\ \alpha \times ($actual $-$ old forecast$)$}}%
  }

\end{frame}

\begin{frame}{The actual grows up: SARSA}
  \centering
  \begin{tikzpicture}
    \useasboundingbox (-1.4,-0.9) rectangle (12.2,1.2);
    \only<1->{%
      \node[text=accent2, font=\bfseries] at (2.4,0.8) {S};
      \node[text=accent2, font=\bfseries] at (4.0,0.8) {A};
      \node[text=accent2, font=\bfseries] at (5.6,0.8) {R};
      \node[text=accent2, font=\bfseries] at (7.2,0.8) {S};
      \node[text=accent2, font=\bfseries] at (8.8,0.8) {A};
      \node[nodebox, minimum width=1.2cm] at (2.4,0) {$s_i$};
      \node[nodebox, minimum width=1.2cm] at (4.0,0) {$a_i$};
      \node[nodebox, minimum width=1.2cm] at (5.6,0) {$r_i$};
      \node[nodebox, minimum width=1.2cm] at (7.2,0) {$s_{i+1}$};
      \node[nodebox, minimum width=1.2cm] at (8.8,0) {$a_{i+1}$};
      \node[text=muted, anchor=west] at (9.8,0) {one logbook line};}%
  \end{tikzpicture}
  \formulaslot{2.0cm}{%
    \only<2>{\Large $q(s_i, a_i) \leftarrow q(s_i, a_i) + \alpha \bigl[\, r_i - q(s_i, a_i) \,\bigr]$ \quad {\normalsize\color{muted}(cash only)}}%
    \only<3>{\Large $q(s_i, a_i) \leftarrow q(s_i, a_i) + \alpha \bigl[\, \underbrace{r_i + \gamma\, q(s_{i+1}, a_{i+1})}_{\text{the actual, grown up}} - \ q(s_i, a_i) \,\bigr]$}%
    \only<4->{\Large $q(s_i, a_i) \leftarrow q(s_i, a_i) + \alpha \underbrace{\bigl[\, r_i + \gamma\, q(s_{i+1}, a_{i+1}) - q(s_i, a_i) \,\bigr]}_{\text{the surprise}}$}%
  }
  \formulaslot{0.7cm}{%
    \uncover<3->{{\color{muted}the future has not happened yet: the guess $q(s_{i+1}, a_{i+1})$ stands in for $Q^\star$}}%
  }
  \formulaslot{0.8cm}{%
    \uncover<5->{\footnotesize $\alpha = 0.15$, constant

\begin{frame}{The Robbins-Monro iteration}
  \formulaslot{0.8cm}{%
    {\color{muted}find the number $x^\star$ that equals the average of noisy samples, one at a time}%
  }
  \formulaslot{1.9cm}{%
    \only<2>{\Large $x \leftarrow x + \alpha_n \, (\text{sample} - x)$}%
    \only<3->{\Large $x \leftarrow x + \underbrace{\alpha_n}_{\text{step } n} \, \underbrace{(\text{sample} - x)}_{\text{the miss}}$}%
  }
  \formulaslot{0.7cm}{%
    \uncover<4->{$\alpha_1 + \alpha_2 + \alpha_3 + \cdots$ unbounded \quad {\color{muted}(can travel from any start)}}%
  }
  \formulaslot{0.7cm}{%
    \uncover<5->{$\alpha_1^2 + \alpha_2^2 + \alpha_3^2 + \cdots$ finite \quad {\color{muted}(the noise eventually cancels)}}%
  }
  \formulaslot{0.7cm}{%
    \uncover<6->{$\alpha_n = 1/n$ satisfies both demands}%
  }
  \takeaway{\uncover<7->{SARSA is Robbins-Monro on the Bellman equation, cell by cell}}
\end{frame}

\begin{frame}{The running average}
  \formulaslot{0.8cm}{%
    {\color{muted}$A_n$: the average of the first $n$ observations $x_1, \dots, x_n$}%
  }
  \formulaslot{1.9cm}{%
    \only<2>{\Large $A_n = A_{n-1} + \tfrac{1}{n} \, \bigl( x_n - A_{n-1} \bigr)$}%
    \only<3->{\Large $A_n = \underbrace{\vphantom{\bigl(} A_{n-1}}_{\text{average so far}} + \ \tfrac{1}{n} \, \underbrace{\bigl( x_n - A_{n-1} \bigr)}_{\text{one share of the surprise}}$}%
  }
  \formulaslot{0.8cm}{%
    \uncover<4->{hold $\alpha$ constant instead of $1/n$: each old week fades by $(1 - \alpha)$}%
  }
  \takeaway{\uncover<5->{SARSA is a running average of the weekly evidence}}
\end{frame}
