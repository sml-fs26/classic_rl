% Roadmap section-divider: HYBRID of last semester's arcade map.
%   * original cartridge art (img/roadmap/*.png, full-res from w3_rl.pptx)
%   * arrows + why-ladder bubbles redrawn natively in TikZ + Libertinus
%   * a "you are here" spotlight: current ride lit, the rest dimmed
%
%   \roadmap            overview: every ride lit, the full why-ladder
%   \roadmap[bellman]   spotlight one ride: mdp | bandit | bellman | darts | sarsa
% ============================================================================
\usetikzlibrary{shadows.blur}
\definecolor{spot}{HTML}{F59E0B}   % bright amber glow for the current ride

% --- one cartridge: #1 file  #2 x  #3 y  #4 lit(1)/dim(0) ---
\newcommand{\rmcard}[4]{%
  \ifnum#4=1
    \node[inner sep=0pt, draw=spot, line width=2.6pt, rounded corners=2.5pt]
      (rmnode) at (#2,#3) {\includegraphics[height=3.15cm]{img/roadmap/#1}};%
  \else
    \begin{scope}[transparency group, opacity=0.26]
      \node[inner sep=0pt] at (#2,#3) {\includegraphics[height=2.95cm]{img/roadmap/#1}};%
    \end{scope}
  \fi
}
% --- a why-ladder bubble: #1 x  #2 y  #3 width(cm)  #4 fill  #5 textcolor  #6 text ---
\newcommand{\rmbubble}[6]{%
  \node[fill=#4, text=#5, rounded corners=3pt, align=center, font=\scriptsize,
        inner sep=4pt, text width=#3cm] at (#1,#2) {#6};%
}

% positions on a 16 x 9 canvas (cards on a U-shaped path, ladder on the right)
\newcommand{\rmscene}[1]{% #1 = current key (or "all")
  \begin{frame}[plain]
  \centering
  \begin{tikzpicture}
    \useasboundingbox (0,0) rectangle (16,9);
    % faint hand-drawn park backdrop
    \node[opacity=0.16, inner sep=0pt] at (5.6,4.5) {\includegraphics[height=9.2cm]{img/roadmap/park}};
    % path arrows (drawn first, behind cards)
    \begin{scope}[draw=accent!75, line width=3pt, -{Stealth[length=4mm]}, opacity=0.9]
      \draw (3.7,6.2) -- (4.1,6.2);
      \draw (6.6,6.2) -- (7.0,6.2);
      \draw (8.2,4.5) -- (8.2,4.1);     % BELLMAN down to DARTS
      \draw (6.9,2.6) -- (6.5,2.6);     % DARTS left to SARSA
    \end{scope}
    % the five cartridges (lit iff current; "all" lights every one)
    \def\cur{#1}
    \rmcard{mdp-maze}{2.4}{6.2}{\ifx\cur\rmmdp 1\else\ifx\cur\rmall 1\else0\fi\fi}
    \rmcard{bandit-arcade}{5.3}{6.2}{\ifx\cur\rmban 1\else\ifx\cur\rmall 1\else0\fi\fi}
    \rmcard{bellman-house}{8.2}{6.2}{\ifx\cur\rmbel 1\else\ifx\cur\rmall 1\else0\fi\fi}
    \rmcard{darts-dark}{8.2}{2.55}{\ifx\cur\rmdar 1\else\ifx\cur\rmall 1\else0\fi\fi}
    \rmcard{sarsa-maze}{5.3}{2.55}{\ifx\cur\rmsar 1\else\ifx\cur\rmall 1\else0\fi\fi}
    % ride labels under each card
    \begin{scope}[font=\scriptsize\bfseries, text=black!70]
      \node at (2.4,4.35) {MDP maze}; \node at (5.3,4.35) {bandit arcade};
      \node at (8.2,4.35) {Bellman's house}; \node at (8.2,0.7) {darts in the dark};
      \node at (5.3,0.7) {SARSA maze};
    \end{scope}
    % the why-ladder (right column)
    \ifx\cur\rmall
      \node[font=\large\bfseries, text=accent] at (12.5,8.4) {The plan};
      \rmbubble{12.5}{7.3}{4.2}{accent}{white}{How do we find the optimal strategy?}
      \rmbubble{12.5}{6.25}{3.9}{good!18!white}{black!80}{the call with the highest cumulative reward}
      \rmbubble{12.5}{5.0}{4.2}{accent}{white}{How do we compute that reward?}
      \rmbubble{12.5}{3.95}{3.9}{good!18!white}{black!80}{the Bellman equation}
      \rmbubble{12.5}{2.7}{4.2}{accent}{white}{How do we solve Bellman without the odds?}
      \rmbubble{12.5}{1.5}{3.9}{good!18!white}{black!80}{Robbins-Monro, i.e. SARSA}
    \else
      \node[font=\large\bfseries, text=spot, anchor=north] at (12.5,8.2) {you are here};
      \rmbubble{12.5}{6.6}{4.3}{accent}{white}{\rmq}
      \rmbubble{12.5}{5.0}{4.0}{good!18!white}{black!80}{\rma}
    \fi
  \end{tikzpicture}
  \end{frame}
}
